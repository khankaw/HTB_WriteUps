#!/usr/bin/env python3
import tarfile
import os
import sys
import re
import argparse

BACKUP_BASE_DIR = "/opt/backup_clients/backups" #Dirección original de los backups
STAGING_BASE = "/opt/backup_clients/restored_backups" #Dirección donde se van a extraer 


#Definir función para validar nombre

def validate_backup_name(filename):
    if not re.fullmatch(r"^backup_\d+\.tar$", filename): #Aquí se valida cierto formato para el nombre del archivo. Tiene que coincidir exactamente con el formato
        return False
    client_id = filename.split('_')[1].rstrip('.tar') #Extraer el número
    return client_id.isdigit() and client_id != "0" #Regresa true si es un número distinto de 0

#Válida que en el tag solo haya números, letras, guión bajo, entre 1 y 24
def validate_restore_tag(tag):
    return bool(re.fullmatch(r"^[a-zA-Z0-9_]{1,24}$", tag))


def main():

#Se definen los argumentos necesarios -b de backup y -r restore-dir para cuando se corra el script con la función argparse. Esto ayuda redefinir los argumentos automáticamente

    parser = argparse.ArgumentParser(
        description="Restore client configuration from a validated backup tarball.",
        epilog="Example: sudo %(prog)s -b backup_1001.tar -r restore_john"
    )
    parser.add_argument(
        "-b", "--backup",
        required=True,
        help="Backup filename (must be in /home/wacky/backup_clients/ and match backup_<client_id>.tar, " #Defie mensaje que muestra la condición del nombre
             "where <client_id> is a positive integer, e.g., backup_1001.tar)"
    )
    parser.add_argument(
        "-r", "--restore-dir",
        required=True,
        help="Staging directory name for the restore operation. "
             "Must follow the format: restore_<client_user> (e.g., restore_john). "
             "Only alphanumeric characters and underscores are allowed in the <client_user> part (1–24 characters)."
    )

    args = parser.parse_args()

    if not validate_backup_name(args.backup): #Si el nombre del backup no es válido -> error y sale
        print("[!] Invalid backup name. Expected format: backup_<client_id>.tar (e.g., backup_1001.tar)", file=sys.stderr)
        sys.exit(1)

    backup_path = os.path.join(BACKUP_BASE_DIR, args.backup) #Se construye la ruta
    if not os.path.isfile(backup_path):
        print(f"[!] Backup file not found: {backup_path}", file=sys.stderr) #Si no es un archivo -> error y sale
        sys.exit(1)

    if not args.restore_dir.startswith("restore_"): #Si el archivo no existe error y sale
        print("[!] --restore-dir must start with 'restore_'", file=sys.stderr)
        sys.exit(1)

    tag = args.restore_dir[8:] #El directorio de restauración debe empezar con restore_
    if not tag: #No debe estar vacío
        print("[!] --restore-dir must include a non-empty tag after 'restore_'", file=sys.stderr) #Si no se cumple condición de empty tag -> error y sale
        sys.exit(1)

    if not validate_restore_tag(tag): #El tag debe cumplir un formato de 1 a 24 caracteres y contener letras, dígitos y guión bajo, si no error y sale
        print("[!] Restore tag must be 1–24 characters long and contain only letters, digits, or underscores", file=sys.stderr)
        sys.exit(1)

    staging_dir = os.path.join(STAGING_BASE, args.restore_dir) #Se construye el path para extracción
    print(f"[+] Backup: {args.backup}")
    print(f"[+] Staging directory: {staging_dir}")

    os.makedirs(staging_dir, exist_ok=True) #Crea el directorio, si existe evita el error

    try:
        with tarfile.open(backup_path, "r") as tar: #Abre tar y extrae el staging dir 
            tar.extractall(path=staging_dir, filter="data") #Extrae de este lugar al otro. Aquí está la vulnerabilidad en la versión de Python que se corre en la máquina
        print(f"[+] Extraction completed in {staging_dir}") 
    except (tarfile.TarError, OSError, Exception) as e:
        print(f"[!] Error during extraction: {e}", file=sys.stderr)
        sys.exit(2)

if __name__ == "__main__":
    main()
